Quick Start

Integration in minutes

Works with any HTTP client. No SDK needed. Try it live with your key.

bash 
# Classify a medicine image
curl -X POST \
  https://your-api.com/v1/detect/analyse \
  -H "X-API-Key: pg_your_key_here" \
  -F "image=@pill.jpg"
response.json 
{
  "success": true,
  "result": {
    "prediction":   "Authentic",
    "confidence":   0.988,
    "is_authentic": true,
    "risk_level":   "LOW",
    "probabilities": {
      "authentic":   0.988,
      "counterfeit": 0.012
    }
  }
}
javascript 
// Node.js / fetch
const form = new FormData();
form.append('image', fs.createReadStream('pill.jpg'));

const res = await fetch('/v1/detect/analyse', {
  method: 'POST',
  headers: { 'X-API-Key': 'pg_your_key' },
  body: form
});
const { result } = await res.json();
